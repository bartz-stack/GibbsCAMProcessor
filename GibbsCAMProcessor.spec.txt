# -*- mode: python ; coding: utf-8 -*-
"""
PyInstaller Spec File for GibbsCAM Processor
---------------------------------------------
This file tells PyInstaller how to build the executable.

To build the executable:
    pyinstaller GibbsCAMProcessor.spec

The executable will be in: dist/GibbsCAMProcessor.exe
"""

import os
from pathlib import Path

# Get the current directory
base_path = Path('.').resolve()
gcp_path = base_path / 'GCP_clau'

# Define data files to include (config and icon)
datas = [
    (str(gcp_path / 'config.ini'), 'GCP_clau'),  # Include config.ini
]

# Add icon if it exists in the project
icon_path = gcp_path / 'Gibbscam.ico'
if icon_path.exists():
    datas.append((str(icon_path), 'GCP_clau'))  # Include icon in bundle
    icon_file = str(icon_path)  # Use as exe icon too
else:
    # Try network path as fallback
    network_icon = Path(r'\\Schuette-DC1\production\GibbsCAM\NPD Setup Sheets\Script Files\Gibbscam.ico')
    if network_icon.exists():
        icon_file = str(network_icon)
    else:
        icon_file = None

# Analysis - find all the Python files
a = Analysis(
    [str(gcp_path / '__main__.py')],
    pathex=[str(base_path)],
    binaries=[],
    datas=datas,
    hiddenimports=[
        'win32com.client',
        'win32gui',
        'win32process',
        'psutil',
        'pandas',
        'openpyxl',
        'winotify',
    ],
    hookspath=[],
    hooksconfig={},
    runtime_hooks=[],
    excludes=[],
    win_no_prefer_redirects=False,
    win_private_assemblies=False,
    cipher=None,
    noarchive=False,
)

# Package everything together
pyz = PYZ(a.pure, a.zipped_data, cipher=None)

# Create the executable
exe = EXE(
    pyz,
    a.scripts,
    a.binaries,
    a.zipfiles,
    a.datas,
    [],
    name='GibbsCAMProcessor',
    debug=False,
    bootloader_ignore_signals=False,
    strip=False,
    upx=True,
    upx_exclude=[],
    runtime_tmpdir=None,
    console=True,  # Set to False to hide console window
    disable_windowed_traceback=False,
    argv_emulation=False,
    target_arch=None,
    codesign_identity=None,
    entitlements_file=None,
    icon=icon_file,  # Exe icon
)
